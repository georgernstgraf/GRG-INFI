-- Lehrer mit Sprechstunde am Montag
SELECT Lehrer.[L_Nr], Lehrer.[L_Zuname], Lehrer.[L_Sprechstunde_tag]
FROM Lehrer
WHERE (((Lehrer.[L_Sprechstunde_tag])="Montag"))
ORDER BY Lehrer.[L_Nr];
	-- oder
SELECT L_Nr, L_Zuname, L_Sprechstunde_tag
FROM Lehrer
WHERE L_Sprechstunde_tag="Montag"
ORDER BY L_Nr

-- Welche Lehrer haben keine Sprechstunde eingetragen
SELECT L_Nr, L_Zuname, L_Sprechstunde_tag
FROM Lehrer
WHERE L_Sprechstunde_tag   IS NULL
ORDER BY L_ZuName

-- Klassen aus Abteilung 3
SELECT K_Nr, K_Abteilung
From Klassen
where K_Abteilung = 3

--Welche Klassen enden vo dem Schuljahresende(haben Inhalt in K_Datumbis)
SELECT K_Nr, K_Abteilung, K_Datumbis
From Klassen
where K_Datumbis is not null

-- welche Klassen gehören zur Fachschule
SELECT K_Nr, K_Abteilung, K_Vorstand
From Klassen
where K_Nr like "??FID"

-- welche Klassen sind 1. jahrgänge
SELECT K_Nr, K_Abteilung, K_Vorstand
From Klassen
where K_Nr like "1*"

-- welche Klassen haben Skolud als KV (man kann auch ein anderes Lehrerkürzel nehmen)
SELECT K_Nr, K_Abteilung, K_Vorstand
From Klassen
where K_Vorstand = 'SKo'

--Welche Schuljahre beginnen im Jahr 2009
SELECT *
From Schuljahre
where year(Sja_Datumvon) = 2009

--Wie lange dauern die verschiedenen Schuljahre (es sind Semester dabei)
SELECT Sja_Bezeichnung, Sja_datumvon, sja_Datumbis,
           Sja_datumbis - Sja_Datumvon as Dauer
From Schuljahre

-- Schüler aus der  HBG  sortiert
SELECT S_Zuname, S_Vorname, S_klasse
From Schueler
where s_klasse like '*HBG*'
order by  s_Klasse, S_Zuname, S_Vorname

--Welche Schüler der Abteilung HIF haben den Vornamen Michael (oder anderen nehmen)
SELECT S_Zuname, S_Vorname, S_klasse
From Schueler
where s_klasse like '*HIF*' and S_Vorname = 'Michael'
order by  s_Klasse, S_Zuname, S_Vorname

-- Welche Schüler wohnen an einem "weg" in den Bezirken 2,12 oder 22 (oder auch 21,23)
SELECT S_Zuname, S_Vorname, S_klasse, S_Strasse, S_postleitzahl
From Schueler
where s_postleitzahl in (1020,1120,1220) and S_Strasse like '*weg*'

-- Welche Schüler wohnen nicht in Wien und sind vor 1993 geboren
SELECT S_Zuname, S_Vorname, S_klasse, S_Strasse, S_postleitzahl, S_Gebdatum
From Schueler
where s_postleitzahl >= 2000 and  year(s_gebdatum) < 1993
