SELECT Artikelnummer, Artikel.Einkaufspreis, Verkaufspreis * 1.2 AS Brutto, Verkaufspreis - einkaufspreis AS Handelsspanne
FROM Artikel
WHERE einkaufspreis < 100 and
         verkaufspreis - einkaufspreis  > 0;
         


select  Name, Postleitzahl, Ort,  Region
   from Kunden

 select  Name, Postleitzahl, Ort,  Region
   from Kunden
 where Postleitzahl between 4000 and 5000
 
select  Name, Postleitzahl, Ort,  Region
   from Kunden
 where Name like 'Fa*'

   from Kunden
 where Name like "*Gmbh*"
           
           
SELECT Artikel.Artikelnummer, Artikel.Einkaufspreis
FROM Artikel
WHERE (((Artikel.Einkaufspreis)<45));
           

SELECT 	Artikelnummer, 
	Artikel.Einkaufspreis, 
	Verkaufspreis * 1.2    AS Brutto,		-- mit  As Brutto wird eine „Überschrift“ vergeben
	Verkaufspreis - einkaufspreis AS Handelsspanne               
WHERE  einkaufspreis < 100 and
             verkaufspreis - einkaufspreis  > 0

SELECT Left([Name],3) AS Ausdr1, Mid([Region],8) AS Ausdr2, Right([Ort],1) AS Ausdr3,
            instr(Region,"SÜD") as AnwelcherStelle
FROM Kunden;
